Basic geometry functions written in Plan9 C.

double
rad2deg(double rad)
{
	return rad * (180/PI);
}

double
deg2rad(double deg)
{
	return deg * (PI/180);
}

double
ptangledeg(Point a, Point b)
{
	double theta = atan2(b.y - a.y,b.x - a.x);
	return rad2deg(theta);
}

double
ptangle(Point a, Point b)
{
	return atan2(b.y - a.y,b.x - a.x);
}

double
ptdistance(Point a, Point b)
{
	double x = a.x - b.x;
	double y = a.y - b.y;
	return sqrt(x*x+y*y);
}

Point
rotpt(Point center, Point p0, double deg)
{
	double rot = ptangle(center, p0) + deg2rad(deg);
	double radius = ptdistance(center, p0);
	return (Point){
		center.x + radius * cos(rot),
		center.y + radius * sin(rot)};
}
